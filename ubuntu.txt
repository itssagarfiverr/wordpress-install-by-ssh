

# Download wp_install.sh script and place it in the /intechost/cloud directory
mkdir -p /intechost/cloud
curl -o /intechost/cloud/wp_install.sh https://raw.githubusercontent.com/itssagarfiverr/wordpress-install-by-ssh/main/wp_install.sh

# Disable other MOTD scripts
sudo chmod -x /etc/update-motd.d/*

# Ensure the setup script is executable and then run it
chmod +x /intechost/cloud/wp_install.sh
cd /intechost/cloud
./wp_install.sh
