

# Download wordpress.sh script and place it in the /intechost/cloud directory
mkdir -p /intechost/cloud
curl -o /intechost/cloud/wordpress.sh https://raw.githubusercontent.com/itssagarfiverr/wordpress-install-by-ssh/main/wordpress.sh

# Disable other MOTD scripts
sudo chmod -x /etc/update-motd.d/*

# Delete default index.html

rm -rf /var/www/html/index.html

# Ensure the setup script is executable and then run it
chmod +x /intechost/cloud/wordpress.sh
/intechost/cloud/wordpress.sh
